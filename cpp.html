Tree
Explanation: Demonstrates Inorder, Preorder, and Postorder tree traversals.

// Inorder Traversal void inorder(Node* root){ if(!root) return; inorder(root->left); cout << root->data; inorder(root->right); }

// Preorder Traversal void preorder(Node* root){ if(!root) return; cout << root->data; preorder(root->left); preorder(root->right); }

// Postorder Traversal void postorder(Node* root){ if(!root) return; postorder(root->left); postorder(root->right); cout << root->data; }
